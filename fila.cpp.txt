struct No {
	int info;
	struct No *prox;

void enfileirar(int valor, struct No *ptr);
void exibirFila();

struct No *cabeca=NULL;
struct No *ult;

int main(){
	struct No *novo = (struct No*)malloc(sizeof(struct No));
	
	enfileirar(10,novo);
	enfileirar(20,novo);
	exibirFila();
	desenfileirar();
	desenfileirar();

void enfileirar(int valor, struct No *ptr){
	struct No *aux;
	aux = cabeca;
	
	struct No *novo = (struct No*)malloc(sizeof(struct No));
	
	aux = novo;

	novo->info = valor; // pega o local do novo e coloca o elem no info 
	novo->prox = NULL; // sempre colocar o proximo prox como null pq ainda não tem o endereço do proximo

	if (cabeca == NULL){
		cabeca = novo; // cabeça recebe o local do do novo
	}
	else{
		ult->prox = novo; // pra onde o ultimo ta apontando vai receber o novo 
	}
	ult = novo; // depois recebe o novo	
}

void exibirFila(){
	struct No *aux;
	aux = cabeca;
	
	if(cabeca == NULL){
		printf("Lista Vazia");
	}
	else{	
		while(aux != NULL){
			printf("%x %i -> %x \n", aux, aux->info, aux->prox); // mostra o local do arquivo a informação e pra onde ta sendo apontado 
			aux = aux->prox;
		}
	}	
}

void desenfileirar(){
	cout << "\n";
	struct No *aux;
	aux = cabeca->prox;
	cabeca = aux;
	free(aux);
}
